 Considerem
$$
\begin{array}{lll}
\mathrm{GL}(n,\mathbb{Z})&:=
\{M \in \mathrm{M}_{n \times n}(\mathbb{Z}):\, \det(M) \in \mathbb{Z}^{\ast}\},
&\text{grup lineal, }\\
\mathrm{SL}(n,\mathbb{Z})&:=
\{M \in \mathrm{GL}(n,\mathbb{Z}):\, \det(M)=1 \},

&\text{grup ortogonal, }\\
\mathrm{SO}(n,\mathbb{Z})&:=
\{M \in \mathrm{O}(n,\mathbb{Z}):\, \det(M)=1 \},
&\text{grup especial ortogonal. }
\end{array}

són subgrups del grup $\mathrm{GL}(n,\mathbb{Z})$.

\end{enumerate}

\begin{ex}[$*$]
{\it Teoremes d'isomorfia de grups.}
\begin{enumerate}[(a)]
\item
Siguin $G$
un grup,
$H$
un subgrup normal i $F$
un subgrup qualsevol.
Proveu que $HF$
és un subgrup de $G$,
que $F\cap H$
és un subgrup normal de $F$,
que $H$
és un subgrup normal de $HF$
i que tenim un isomorfisme de grups
$$
HF/H\simeq F/F\cap H.
$$
\item
Siguin $\varphi:G \rightarrow G'$
un morfisme exhaustiu de grups,
$H'$
un subgrup normal de $G'$
i $H=\varphi^{-1}(H')$.
Proveu que $\varphi$
indueix un isomorfisme entre $G/H$
i $G'/H'$.
\item
Siguin $G$
un grup i $F\subset H$
dos subgrups normals de $G$.
Demostreu que $H/F$
és un subgrup normal de $G/F$
i que tenim un isomorfisme de grups
$$
(G/F) / (H/F) \simeq G/H.
$$
\end{enumerate}
\end{ex}
