{
	\introduction
	{ 
		$introduction\\$ 
	}

	\proposition{ $ Characterization of regular exponential models $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ 1-D exponential model parametrized by $ \Theta.
			\exp( f(x)\phi(\theta) + f'(x) + \phi'(\theta) ) $ likelihood function of $ ( \Omega, \Ac, \Pc ) $ statistical model $
		}
		\holds
		{
			\Theta $ interval $, \phi,\phi' \in \Cc^2.
			\all{ \theta \in \Theta }
			{
				\phi'(\theta) \neq 0.
				E_\theta f^2(x) \in \R
			}.

			\imp ( \Omega, \Ac, \Pc ) $ regular $
		}
		\demonstration
		{
			$ no demonstration $
		}
	}

}
