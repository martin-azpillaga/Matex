\begin{ex}
Siguin $A$
un anell i $f$,
$g\in A[X]$.
\begin{enumerate}[(a)]
\item
Demostreu que si $g$
és mònic,
aleshores existeixen polinomis $q$,
$r\in A[X]$
tals que $f=gq+r$,
amb $r=0$
o bé $gr(r)<gr(g)$.
Demostreu que $q$
i $r$
són únics per a aquestes condicions.
Una igualtat $f=gq+r$
s'anomena la {\it divisió entera}
de $f$
per $g$
si $r=0$
o bé $gr(r)<gr(g)$;
els polinomis $q$
i $r$
s'anomenen el {\it quocient}
i el {\it residu}
de la divisió entera.
Si el residu de la divisió entera de $f$
per $g$
és $0$,
es té que $f$
és un múltiple de $g$
i que $g$
és un divisor de $f$.
\item
Doneu un exemple de no-unicitat i un de no-existència en el cas en què $g$
no sigui mònic.
\end{enumerate}
\end{ex}

\begin{ex}[$*$]
\label{nil}
Sigui $A$
un anell.
Es diu que un element $x \in A$
és {\em nilpotent}
si $x^n = 0$
per a algun nombre enter positiu $n$.
Proveu que
\begin{enumerate}[(a)]
\item
Si $x$
és nilpotent,
aleshores $1-x$
és invertible.
\item
El conjunt d'elements nilpotents de $A$
és un ideal radical,
que notem $\eta(A)$
i s'anomena nilradical de $A$.
\item
$\eta (A)$
està contingut dins de tots els ideals primers de $A$.
\item
$\eta(A)$
és la intersecció de tots els ideals primers de $A$.
({\em Indicació:}
Sigui $y\notin \eta(A)$
i definim $\Sigma$
com el conjunt d'ideals $K\subseteq A$
tals que $y^n\notin K$
per a tot $n>0$.
Vegeu que $\Sigma$
té un element maximal i que aquest és un ideal primer.)
\item
Sigui $J$
un ideal de $A$.
Proveu que el radical de $J$
és la intersecció de tots els ideals primers de $A$
que contenen $J$.
\end{enumerate}
\end{ex}

\begin{ex}[$*$]
Siguin $d$
un nombre enter lliure de quadrats i
$A=\mathbb{Z}[\sqrt{d}]=\{a+b\sqrt{d}: a,b\in \mathbb{Z}\} \subseteq \mathbb{C}.$
\begin{enumerate}[(a)]
\item
A partir de la igualtat $(d+\sqrt{d})(d-\sqrt{d})=d(d-1)$,
proveu que $2$
no és un element primer de l'anell $A$.
Deduïu que,
si $A$
és un domini de factorització única,
$2$
no és irreductible a $A$.
\item
Demostreu que,
si $d$
és negatiu,
$A$
és un domini de factorització única si,
i només si,
$d=-1$
o $d=-2$.
\item
Proveu que $5$
és irreductible a $\mathbb{Z}\left[\sqrt{-2}\,\right]$
i que l'anell quocient $\mathbb{Z}\left[\sqrt{-2}\,\right]/(5)$
té un subcòs isomorf a $\mathbb{Z} /5\mathbb{Z}$.
\end{enumerate}
\end{ex}


\begin{ex}
Es considera l'anell $\mathbb{Z}[\sqrt{d}\,]$,
on $d \in \mathbb{Z}$
no és un quadrat perfecte.
Sigui $x+\sqrt{d} \in \mathbb{Z}[\sqrt{d}\,]$,
$x\in\mathbb{Z}$,
un element de norma $N(x+\sqrt{d})=n$.
\begin{enumerate}[(a)]
\item
Demostreu que l'aplicació
$$
\begin{array}{llcl}
f: & \mathbb{Z}[\sqrt{d}\,] &\rightarrow & \mathbb{Z}/n\mathbb{Z}    \\
   & a+b\sqrt{d}                 & \mapsto     & \overline{a-xb}
\end{array}
$$
és un morfisme d'anells que factoritza en un isomorfisme
$\overline{f}: \mathbb{Z}[\sqrt{d}\,]/(x+\sqrt{d})\simeq \mathbb{Z}/n\mathbb{Z}$.

\noindent
Sigui $d=15$
i considerem l'anell $\mathbb{Z}\left[\sqrt{15}\,\right]$.

\item
Esbrineu si els elements $1+\sqrt{15}$,
$2+\sqrt{15}$
són primers.
Són irreductibles?

\noindent
{\textit{Indicació}:
Demostreu que l'equació $z^2=\pm 2$
no té solucions en $\mathbb{Z}/5\mathbb{Z}$.}

\item
És $\mathbb{Z}\left[\sqrt{15}\,\right]$
un domini de factorització única?
\end{enumerate}
\end{ex}
