\unit{ $ Jordan form $ }
{
	
	\definition{ $ Eigenvalue $ }
	{
		\letbe
		{
			A \in \Mc_{n \times n}(K).
			\lambda \in K	
		}
		\then{ \lambda }{ $ an eigenvalue $ }
		{
			\ex{ v \in K \nonzero }
			{
				Av = \lambda v
			}
		}
		\denote
		{
			\set{ \lambda \in K \with \lambda $ eigenvalue $ } \as Spec(A)
		}
	}


	\definition{ $ Eigenvector $ }
	{
		\letbe
		{
			A \in \Mc_{n \times n}(K).
			\lambda \in K $ eigenvalue $.
			v \in K
		}
		\then{ v }{ $ an eigenvector of eigenvalue $ \lambda }
		{
			Av = \lambda v
		}
		\denote
		{
			\set{ v \in K \with v $ eigenvector of eigenvalue $ \lambda } \as Ker_\lambda(K)
		}
	}

}
